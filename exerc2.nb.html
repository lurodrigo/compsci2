euler = function(f, y0, a, b, h) {
  ts = seq(a, b, by = h)
  
  y = matrix(0, nrow = length(ts), ncol = length(y0))
  y[1,] = y0
  for (i in 1:(length(ts) - 1)) {
    y[i + 1,] = y[i,] + h*f(ts[i], y[i,])
  }
  
  m = cbind(ts, y)
  colnames(m) = c("t", if (is.null(names(y0))) paste0("y", 1:length(y0)) else names(y0))
  m
}
